def frequency_counter(arr: List[int]) -> Dict[int, int]:
    freq = {}
    for x in arr:
        freq[x] = freq.get(x, 0) + 1
    return freq

### 2ï¸âƒ£ Two Sum

> Return indices of two numbers that add up to target

```python
def two_sum(arr: List[int], target: int) -> List[int]:
    seen = {}

    for i, num in enumerate(arr):
        needed = target - num
        if needed in seen:
            return [seen[needed], i]
        seen[num] = i

    return []
```

### 3ï¸âƒ£ Group Anagrams

```python
def group_anagrams(words: List[str]) -> List[List[str]]:
    groups = defaultdict(list)

    for word in words:
        key = tuple(sorted(word))
        groups[key].append(word)

    return list(groups.values())
```

ðŸ‘‰ Faster (26-char hash, lowercase):

```python
def group_anagrams_fast(words: List[str]) -> List[List[str]]:
    groups = defaultdict(list)

    for word in words:
        count = [0] * 26
        for ch in word:
            count[ord(ch) - ord('a')] += 1
        groups[tuple(count)].append(word)

    return list(groups.values())
```

### 4ï¸âƒ£ Longest Consecutive Sequence

```python
def longest_consecutive(nums: List[int]) -> int:
    num_set = set(nums)
    longest = 0

    for num in num_set:
        if num - 1 not in num_set:  # start of sequence
            length = 1
            current = num

            while current + 1 in num_set:
                current += 1
                length += 1

            longest = max(longest, length)

    return longest
```

### 5ï¸âƒ£ Find Duplicates

```python
def find_duplicates(arr: List[int]) -> List[int]:
    seen = set()
    duplicates = set()

    for x in arr:
        if x in seen:
            duplicates.add(x)
        else:
            seen.add(x)

    return list(duplicates)
```

## ðŸ§  Hashing Pattern Recognition

Use **hashing** when:

* You need **O(1)** lookups
* Counting or grouping is involved
* Problem says *pair*, *frequency*, *distinct*, *duplicate*
* Order doesnâ€™t matter
