### 1ï¸âƒ£ Reverse an Array

```python
def reverse_array(arr: List[int]) -> List[int]:
    return arr[::-1]
```

### 2ï¸âƒ£ Rotate Array by k Positions (Right Rotation)

```python
def rotate_array(arr: List[int], k: int) -> List[int]:
    n = len(arr)
    if n == 0:
        return arr
    k %= n
    return arr[-k:] + arr[:-k]
```

ðŸ‘‰ For **left rotation**, swap the slices:

```python
def rotate_array_left(arr: List[int], k: int) -> List[int]:
    n = len(arr)
    if n == 0:
        return arr
    k %= n
    return arr[k:] + arr[:k]
```

### 3ï¸âƒ£ Find Max and Min in Array

```python
def find_max_min(arr: List[int]) -> Tuple[int, int]:
    if not arr:
        raise ValueError("Array is empty")
    return max(arr), min(arr)
```

### 4ï¸âƒ£ Remove Duplicates (Preserve Order)

```python
def remove_duplicates(arr: List[int]) -> List[int]:
    seen = set()
    result = []
    for x in arr:
        if x not in seen:
            seen.add(x)
            result.append(x)
    return result
```

---

### 5ï¸âƒ£ Check if Array is Sorted (Ascending)

```python
def is_sorted(arr: List[int]) -> bool:
    return all(arr[i] <= arr[i + 1] for i in range(len(arr) - 1))
```

### ðŸ§ª Example Usage

```python
arr = [3, 1, 2, 2, 4]

print(reverse_array(arr))          # [4, 2, 2, 1, 3]
print(rotate_array(arr, 2))        # [2, 4, 3, 1, 2]
print(find_max_min(arr))           # (4, 1)
print(remove_duplicates(arr))      # [3, 1, 2, 4]
print(is_sorted(arr))              # False
```
